详情页的滚动效果：全屏竖向切换

为我的个人主页项目详情页开发全屏竖向滚动切换动效组件，仅实现动效功能，遵循以下要求：

1. 技术栈：Next.js + React + TypeScript + Tailwind CSS

2. 动效容器：改为整个浏览器页面全屏，页面切换铺满视口

3. 页面内容规则：

  - 第1个页面：保留没给页面的标题、标签、描述文本完整展示

  - 第2-5个页面：仅展示一个标题文字，无任何其他内容

  - 总共制作5个页面的切换示例

4. 核心动效：保留原有的竖向滚动卡片视差切换效果

5. 增加功能：

  - 竖向排列的小圆点轮播指示器

  - 指示器固定在页面侧边，跟随页面滚动同步高亮当前页

6. 第2-5个页面先不展示内容，仅保留动效逻辑

7. 提供完整可直接使用的组件代码 + 页面调用示例代码

1. 组件文件：/components/ui/scroll-cards.tsx

"use client";
import { FC, useEffect, useState } from "react";

// 类型定义
interface iCardItem {
  title: string;
  description?: string;
  tag?: string;
  color: string;
  textColor: string;
}

interface iCardProps extends iCardItem {
  index: number;
}

// 卡片组件
const Card: FC<iCardProps> = ({ title, description, tag, color, textColor, index }) => {
  // 仅第一页展示完整内容，其他页面只显示标题
  const isFirstPage = index === 0;

  return (
    <div 
      className="h-screen w-full flex flex-col items-center justify-center sticky top-0"
      style={{ backgroundColor: color }}
    >
      {/* 标题（所有页面都展示） */}
      <h1 
        className="text-4xl md:text-6xl font-bold mb-4"
        style={{ color: textColor }}
      >
        {title}
      </h1>

      {/* 仅第一页展示标签和描述 */}
      {isFirstPage && (
        <>
          <span 
            className="px-3 py-1 rounded-full text-sm mb-4"
            style={{ backgroundColor: "rgba(0,0,0,0.1)", color: textColor }}
          >
            {tag}
          </span>
          <p 
            className="text-lg md:text-xl max-w-xl text-center"
            style={{ color: textColor }}
          >
            {description}
          </p>
        </>
      )}
    </div>
  );
};

// 轮播指示器组件
const PageIndicator: FC<{ total: number; current: number }> = ({ total, current }) => {
  return (
    <div className="fixed right-6 top-1/2 -translate-y-1/2 flex flex-col gap-3 z-50">
      {Array.from({ length: total }).map((_, i) => (
        <div
          key={i}
          className={`w-3 h-3 rounded-full transition-all ${
            current === i ? "bg-white scale-125" : "bg-gray-400"
          }`}
        />
      ))}
      <p className="text-white text-sm mt-2 text-center">
        {current + 1} / {total}
      </p>
    </div>
  );
};

// 主滚动组件
const CardsParallax: FC<{ items: iCardItem[] }> = ({ items }) => {
  const [currentPage, setCurrentPage] = useState(0);

  // 监听滚动，更新当前页码
  useEffect(() => {
    const handleScroll = () => {
      const page = Math.round(window.scrollY / window.innerHeight);
      setCurrentPage(Math.min(page, items.length - 1));
    };
    window.addEventListener("scroll", handleScroll);
    return () => window.removeEventListener("scroll", handleScroll);
  }, [items.length]);

  return (
    <div className="w-full">
      {/* 指示器 */}
      <PageIndicator total={items.length} current={currentPage} />
      
      {/* 滚动页面 */}
      {items.map((item, i) => (
        <Card key={i} {...item} index={i} />
      ))}
    </div>
  );
};

export { CardsParallax, type iCardItem };
